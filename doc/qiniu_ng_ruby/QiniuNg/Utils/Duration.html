def initialize(args = 0)
  if args.is_a?(Hash)
    @seconds = 0
    MULTIPLES.each do |unit, multiple|
      unit = unit.to_sym
      @seconds += args[unit] * multiple if args.key?(unit)
    end
  else
    @seconds = args.to_i
  end
  calculate!
end
